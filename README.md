## 🔐 Smart Contracts

Flock leverages smart contracts to ensure transparency and automation in collective decision-making.

```solidity
// Example Smart Contract (Solidity)
pragma solidity ^0.8.0;

contract FlockConsensus {
    mapping(address => uint256) public votes;

    function vote(uint256 value) public {
        votes[msg.sender] = value;
    }

    function getVote(address user) public view returns (uint256) {
        return votes[user];
    }
}
```
